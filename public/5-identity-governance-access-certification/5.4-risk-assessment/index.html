1.1 Risk Categories Definition

import boto3
import json
from datetime import datetime, timedelta
from enum import Enum

class RiskCategory(Enum):
    PRIVILEGE_ESCALATION = "privilege_escalation"
    UNAUTHORIZED_ACCESS = "unauthorized_access"
    DATA_EXPOSURE = "data_exposure"
    COMPLIANCE_VIOLATION = "compliance_violation"
    OPERATIONAL_RISK = "operational_risk"

class RiskSeverity(Enum):
    CRITICAL = 10
    HIGH = 8
    MEDIUM = 5
    LOW = 3
    MINIMAL = 1

class RiskAssessmentFramework:
    def __init__(self):
        self.risk_rules = self.load_risk_rules()
        self.threat_intelligence = self.load_threat_intelligence()
        self.compliance_requirements = self.load_compliance_requirements()
    
    def load_risk_rules(self):
        """Load risk assessment rules"""
        return {
            "privilege_escalation": [
                {
                    "rule_id": "PE001",
                    "name": "Administrative Access Without MFA",
                    "description": "Users with admin privileges not using MFA",
                    "severity": RiskSeverity.CRITICAL,
                    "detection_logic": self.detect_admin_no_mfa
                },
                {
                    "rule_id": "PE002", 
                    "name": "Excessive Permissions",
                    "description": "Users with more permissions than required",
                    "severity": RiskSeverity.HIGH,
                    "detection_logic": self.detect_excessive_permissions
                }
            ],
            "unauthorized_access": [
                {
                    "rule_id": "UA001",
                    "name": "Access from Unusual Locations",
                    "description": "Access from geographic locations outside normal patterns",
                    "severity": RiskSeverity.HIGH,
                    "detection_logic": self.detect_unusual_locations
                },
                {
                    "rule_id": "UA002",
                    "name": "Off-Hours Access",
                    "description": "Access during non-business hours",
                    "severity": RiskSeverity.MEDIUM,
                    "detection_logic": self.detect_off_hours_access
                }
            ],
            "data_exposure": [
                {
                    "rule_id": "DE001",
                    "name": "Overprivileged S3 Access",
                    "description": "Broad S3 permissions that could expose sensitive data",
                    "severity": RiskSeverity.HIGH,
                    "detection_logic": self.detect_overprivileged_s3
                }
            ],
            "compliance_violation": [
                {
                    "rule_id": "CV001",
                    "name": "SOX Compliance Violation",
                    "description": "Access patterns violating SOX requirements",
                    "severity": RiskSeverity.CRITICAL,
                    "detection_logic": self.detect_sox_violations
                }
            ]
        }
    
    def assess_risks(self, identity_data):
        """Perform comprehensive risk assessment"""
        
        assessment_results = {
            "assessment_id": f"risk_assessment_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
            "timestamp": datetime.now().isoformat(),
            "total_risks": 0,
            "risk_summary": {
                "critical": 0,
                "high": 0,
                "medium": 0,
                "low": 0
            },
            "risks_by_category": {},
            "detailed_findings": [],
            "remediation_plan": []
        }
        
        # Run all risk rules
        for category, rules in self.risk_rules.items():
            category_risks = []
            
            for rule in rules:
                findings = rule["detection_logic"](identity_data)
                
                for finding in findings:
                    risk_item = {
                        "risk_id": f"{rule['rule_id']}_{len(assessment_results['detailed_findings'])}",
                        "rule_id": rule["rule_id"],
                        "category": category,
                        "name": rule["name"],
                        "description": rule["description"],
                        "severity": rule["severity"].name,
                        "severity_score": rule["severity"].value,
                        "affected_resource": finding["resource"],
                        "evidence": finding["evidence"],
                        "business_impact": self.calculate_business_impact(rule, finding),
                        "remediation_effort": self.estimate_remediation_effort(rule, finding),
                        "detected_at": datetime.now().isoformat()
                    }
                    
                    assessment_results["detailed_findings"].append(risk_item)
                    category_risks.append(risk_item)
                    
                    # Update counters
                    assessment_results["total_risks"] += 1
                    severity_key = rule["severity"].name.lower()
                    assessment_results["risk_summary"][severity_key] += 1
            
            assessment_results["risks_by_category"][category] = category_risks
        
        # Generate remediation plan
        assessment_results["remediation_plan"] = self.generate_remediation_plan(
            assessment_results["detailed_findings"]
        )
        
        return assessment_results
    
    def detect_admin_no_mfa(self, identity_data):
        """Detect admin users without MFA"""
        findings = []
        
        for user in identity_data.get("iam_data", {}).get("users", []):
            # Check if user has admin policies
            admin_policies = [
                "arn:aws:iam::aws:policy/AdministratorAccess",
                "arn:aws:iam::aws:policy/PowerUserAccess"
            ]
            
            has_admin_access = any(
                policy in user.get("attached_policies", []) 
                for policy in admin_policies
            )
            
            if has_admin_access:
                # Check MFA status
                mfa_enabled = self.check_user_mfa_status(user["username"])
                
                if not mfa_enabled:
                    findings.append({
                        "resource": user["username"],
                        "evidence": {
                            "admin_policies": [p for p in user.get("attached_policies", []) if p in admin_policies],
                            "mfa_enabled": False,
                            "last_login": user.get("password_last_used")
                        }
                    })
        
        return findings
    
    def detect_excessive_permissions(self, identity_data):
        """Detect users with excessive permissions"""
        findings = []
        
        for user in identity_data.get("iam_data", {}).get("users", []):
            # Calculate permission score
            permission_score = self.calculate_permission_score(user)
            usage_score = self.calculate_usage_score(user)
            
            # If permissions significantly exceed usage, flag as excessive
            if permission_score > usage_score * 2 and permission_score > 7:
                findings.append({
                    "resource": user["username"],
                    "evidence": {
                        "permission_score": permission_score,
                        "usage_score": usage_score,
                        "attached_policies": user.get("attached_policies", []),
                        "last_activity": user.get("last_activity")
                    }
                })
        
        return findings
    
    def detect_unusual_locations(self, identity_data):
        """Detect access from unusual geographic locations"""
        findings = []
        
        # This would analyze CloudTrail data for geographic anomalies
        # Simplified implementation
        for event in identity_data.get("usage_data", {}).get("suspicious_patterns", []):
            if event.get("type") == "external_ip_access":
                findings.append({
                    "resource": event["user"],
                    "evidence": {
                        "source_ip": event["source_ip"],
                        "event_time": event["event_time"],
                        "location": self.get_ip_location(event["source_ip"])
                    }
                })
        
        return findings
    
    def detect_off_hours_access(self, identity_data):
        """Detect access during off-hours"""
        findings = []
        
        for event in identity_data.get("usage_data", {}).get("suspicious_patterns", []):
            if event.get("type") == "off_hours_access":
                findings.append({
                    "resource": event["user"],
                    "evidence": {
                        "event_time": event["event_time"],
                        "source_ip": event["source_ip"],
                        "business_hours": "09:00-17:00 UTC"
                    }
                })
        
        return findings
    
    def detect_overprivileged_s3(self, identity_data):
        """Detect overprivileged S3 access"""
        findings = []
        
        dangerous_s3_permissions = [
            "s3:*",
            "s3:GetObject",
            "s3:PutObject",
            "s3:DeleteObject"
        ]
        
        for user in identity_data.get("iam_data", {}).get("users", []):
            # Check policies for broad S3 permissions
            for policy_arn in user.get("attached_policies", []):
                policy_doc = self.get_policy_document(policy_arn)
                
                if self.has_broad_s3_permissions(policy_doc):
                    findings.append({
                        "resource": user["username"],
                        "evidence": {
                            "policy_arn": policy_arn,
                            "s3_permissions": self.extract_s3_permissions(policy_doc),
                            "s3_usage": self.get_s3_usage_stats(user["username"])
                        }
                    })
        
        return findings
    
    def detect_sox_violations(self, identity_data):
        """Detect SOX compliance violations"""
        findings = []
        
        # SOX requires segregation of duties
        sox_sensitive_permissions = [
            "iam:CreateUser",
            "iam:DeleteUser", 
            "iam:AttachUserPolicy",
            "organizations:CreateAccount"
        ]
        
        for user in identity_data.get("iam_data", {}).get("users", []):
            # Check if user has multiple conflicting permissions
            user_permissions = self.get_all_user_permissions(user)
            
            conflicting_permissions = [
                perm for perm in user_permissions 
                if perm in sox_sensitive_permissions
            ]
            
            if len(conflicting_permissions) > 1:
                findings.append({
                    "resource": user["username"],
                    "evidence": {
                        "conflicting_permissions": conflicting_permissions,
                        "sox_requirement": "Segregation of duties",
                        "violation_type": "Multiple sensitive permissions"
                    }
                })
        
        return findings
    
    def calculate_business_impact(self, rule, finding):
        """Calculate business impact score"""
        
        base_impact = {
            RiskSeverity.CRITICAL: 10,
            RiskSeverity.HIGH: 8,
            RiskSeverity.MEDIUM: 5,
            RiskSeverity.LOW: 3,
            RiskSeverity.MINIMAL: 1
        }
        
        # Adjust based on resource criticality
        resource_multiplier = self.get_resource_criticality_multiplier(
            finding["resource"]
        )
        
        return base_impact[rule["severity"]] * resource_multiplier
    
    def estimate_remediation_effort(self, rule, finding):
        """Estimate effort required for remediation"""
        
        effort_mapping = {
            "PE001": {"hours": 2, "complexity": "LOW"},      # Enable MFA
            "PE002": {"hours": 8, "complexity": "MEDIUM"},   # Review permissions
            "UA001": {"hours": 4, "complexity": "MEDIUM"},   # IP restrictions
            "UA002": {"hours": 2, "complexity": "LOW"},      # Time restrictions
            "DE001": {"hours": 12, "complexity": "HIGH"},    # S3 policy review
            "CV001": {"hours": 16, "complexity": "HIGH"}     # SOX remediation
        }
        
        return effort_mapping.get(rule["rule_id"], {"hours": 4, "complexity": "MEDIUM"})
    
    def generate_remediation_plan(self, findings):
        """Generate prioritized remediation plan"""
        
        # Sort by severity and business impact
        sorted_findings = sorted(
            findings,
            key=lambda x: (x["severity_score"], x["business_impact"]),
            reverse=True
        )
        
        remediation_plan = []
        
        for i, finding in enumerate(sorted_findings[:20]):  # Top 20 risks
            remediation_item = {
                "priority": i + 1,
                "risk_id": finding["risk_id"],
                "title": finding["name"],
                "affected_resource": finding["affected_resource"],
                "recommended_actions": self.get_remediation_actions(finding),
                "estimated_effort": finding["remediation_effort"],
                "business_impact": finding["business_impact"],
                "target_completion": self.calculate_target_date(finding)
            }
            
            remediation_plan.append(remediation_item)
        
        return remediation_plan
    
    def get_remediation_actions(self, finding):
        """Get specific remediation actions for a finding"""
        
        actions_map = {
            "PE001": [
                "Enable MFA for the user account",
                "Enforce MFA policy for admin users",
                "Review admin access necessity"
            ],
            "PE002": [
                "Review user's actual permission usage",
                "Remove unused permissions",
                "Implement least privilege principle"
            ],
            "UA001": [
                "Implement IP-based access restrictions",
                "Review VPN access policies",
                "Enable geographic access controls"
            ],
            "DE001": [
                "Review S3 bucket policies",
                "Implement bucket-level restrictions",
                "Enable S3 access logging"
            ]
        }
        
        rule_id = finding["rule_id"]
        return actions_map.get(rule_id, ["Review and remediate the identified risk"])

def lambda_handler(event, context):
    """Lambda handler for risk assessment"""
    
    framework = RiskAssessmentFramework()
    
    # Get identity data from previous steps
    identity_data = event.get("identity_data", {})
    
    # Perform risk assessment
    assessment_results = framework.assess_risks(identity_data)
    
    # Store results in S3
    s3_client = boto3.client('s3')
    bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
    key = f'risk-assessments/{datetime.now().strftime("%Y/%m/%d")}/assessment-{assessment_results["assessment_id"]}.json'
    
    s3_client.put_object(
        Bucket=bucket_name,
        Key=key,
        Body=json.dumps(assessment_results, indent=2, default=str),
        ContentType='application/json'
    )
    
    # Send high-priority alerts
    if assessment_results["risk_summary"]["critical"] > 0:
        send_critical_risk_alert(assessment_results)
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Risk assessment completed',
            'assessment_id': assessment_results['assessment_id'],
            'total_risks': assessment_results['total_risks'],
            'critical_risks': assessment_results['risk_summary']['critical'],
            'report_location': f's3://{bucket_name}/{key}'
        })
    }

def send_critical_risk_alert(assessment_results):
    """Send alert for critical risks"""
    
    sns_client = boto3.client('sns')
    
    critical_risks = [
        risk for risk in assessment_results['detailed_findings']
        if risk['severity'] == 'CRITICAL'
    ]
    
    message = f"""
    CRITICAL SECURITY RISKS DETECTED
    
    Assessment ID: {assessment_results['assessment_id']}
    Total Critical Risks: {len(critical_risks)}
    
    Top Critical Risks:
    """
    
    for risk in critical_risks[:5]:
        message += f"""
    - {risk['name']}
      Resource: {risk['affected_resource']}
      Impact: {risk['business_impact']}
    """
    
    sns_client.publish(
        TopicArn='arn:aws:sns:region:account:critical-security-alerts',
        Subject='CRITICAL: Security Risks Detected in Identity Governance',
        Message=message
    )

3.1 Real-time Risk Monitor

import boto3
import json
from datetime import datetime

class RealTimeRiskMonitor:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.sns = boto3.client('sns')
        self.dynamodb = boto3.resource('dynamodb')
        self.risk_table = self.dynamodb.Table('RiskMonitoring')
    
    def process_security_event(self, event):
        """Process incoming security events for risk assessment"""
        
        event_type = event.get('eventName', '')
        source_ip = event.get('sourceIPAddress', '')
        user_identity = event.get('userIdentity', {})
        
        # Analyze event for risk indicators
        risk_score = self.calculate_event_risk_score(event)
        
        if risk_score >= 7:  # High risk threshold
            self.handle_high_risk_event(event, risk_score)
        
        # Update risk metrics
        self.update_risk_metrics(event, risk_score)
        
        return {
            'event_processed': True,
            'risk_score': risk_score,
            'action_taken': risk_score >= 7
        }
    
    def calculate_event_risk_score(self, event):
        """Calculate risk score for a security event"""
        
        score = 0
        
        # High-risk event types
        high_risk_events = [
            'CreateUser',
            'AttachUserPolicy',
            'CreateRole',
            'AssumeRole',
            'DeleteUser',
            'PutUserPolicy'
        ]
        
        if event.get('eventName') in high_risk_events:
            score += 5
        
        # External IP access
        source_ip = event.get('sourceIPAddress', '')
        if not self.is_internal_ip(source_ip):
            score += 3
        
        # Off-hours access
        event_time = datetime.fromisoformat(event.get('eventTime', '').replace('Z', '+00:00'))
        if event_time.hour < 6 or event_time.hour > 22:
            score += 2
        
        # Failed events
        if event.get('errorCode'):
            score += 1
        
        # Root user activity
        if event.get('userIdentity', {}).get('type') == 'Root':
            score += 4
        
        return min(score, 10)
    
    def handle_high_risk_event(self, event, risk_score):
        """Handle high-risk security events"""
        
        # Store in DynamoDB for tracking
        self.risk_table.put_item(
            Item={
                'event_id': event.get('eventID', ''),
                'timestamp': datetime.now().isoformat(),
                'risk_score': risk_score,
                'event_name': event.get('eventName', ''),
                'source_ip': event.get('sourceIPAddress', ''),
                'user_identity': json.dumps(event.get('userIdentity', {})),
                'status': 'DETECTED'
            }
        )
        
        # Send immediate alert
        self.send_risk_alert(event, risk_score)
        
        # Trigger automated response if configured
        if risk_score >= 9:  # Critical threshold
            self.trigger_automated_response(event)
    
    def send_risk_alert(self, event, risk_score):
        """Send risk alert via SNS"""
        
        message = f"""
        HIGH RISK SECURITY EVENT DETECTED
        
        Risk Score: {risk_score}/10
        Event: {event.get('eventName', 'Unknown')}
        User: {event.get('userIdentity', {}).get('userName', 'Unknown')}
        Source IP: {event.get('sourceIPAddress', 'Unknown')}
        Time: {event.get('eventTime', 'Unknown')}
        
        Please investigate immediately.
        """
        
        self.sns.publish(
            TopicArn='arn:aws:sns:region:account:high-risk-alerts',
            Subject=f'HIGH RISK: {event.get("eventName", "Security Event")}',
            Message=message
        )
    
    def update_risk_metrics(self, event, risk_score):
        """Update CloudWatch metrics"""
        
        # Put custom metrics
        self.cloudwatch.put_metric_data(
            Namespace='IdentityGovernance/Risk',
            MetricData=[
                {
                    'MetricName': 'RiskScore',
                    'Value': risk_score,
                    'Unit': 'None',
                    'Dimensions': [
                        {
                            'Name': 'EventType',
                            'Value': event.get('eventName', 'Unknown')
                        }
                    ]
                },
                {
                    'MetricName': 'HighRiskEvents',
                    'Value': 1 if risk_score >= 7 else 0,
                    'Unit': 'Count'
                }
            ]
        )

def lambda_handler(event, context):
    """Lambda handler for real-time risk monitoring"""
    
    monitor = RealTimeRiskMonitor()
    
    # Process CloudTrail events
    results = []
    
    for record in event.get('Records', []):
        if 'cloudtrail' in record.get('eventSource', ''):
            # Parse CloudTrail event
            ct_event = json.loads(record['body'])
            result = monitor.process_security_event(ct_event)
            results.append(result)
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'processed_events': len(results),
            'high_risk_events': sum(1 for r in results if r['action_taken'])
        })
    }
