    def execute_audit_cycle(self, audit_scope="full"):
        """Execute complete audit cycle"""
        
        audit_results = {
            "audit_id": f"audit_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
            "audit_date": datetime.now().isoformat(),
            "audit_scope": audit_scope,
            "controls_tested": [],
            "findings": [],
            "summary": {
                "total_controls": 0,
                "passed": 0,
                "failed": 0,
                "exceptions": 0
            }
        }
        
        # Execute tests for each control category
        for category, controls in self.audit_controls.items():
            if audit_scope == "full" or category in audit_scope:
                
                for control in controls:
                    try:
                        # Execute control test
                        test_result = control["test_procedure"](control)
                        
                        audit_results["controls_tested"].append({
                            "control_id": control["control_id"],
                            "control_name": control["control_name"],
                            "category": category,
                            "test_result": test_result["status"],
                            "findings": test_result.get("findings", []),
                            "evidence": test_result.get("evidence", {}),
                            "test_date": datetime.now().isoformat()
                        })
                        
                        # Update summary
                        audit_results["summary"]["total_controls"] += 1
                        
                        if test_result["status"] == "PASS":
                            audit_results["summary"]["passed"] += 1
                        elif test_result["status"] == "FAIL":
                            audit_results["summary"]["failed"] += 1
                            audit_results["findings"].extend(test_result.get("findings", []))
                        else:
                            audit_results["summary"]["exceptions"] += 1
                        
                    except Exception as e:
                        # Handle test execution errors
                        audit_results["controls_tested"].append({
                            "control_id": control["control_id"],
                            "control_name": control["control_name"],
                            "category": category,
                            "test_result": "ERROR",
                            "error": str(e),
                            "test_date": datetime.now().isoformat()
                        })
                        
                        audit_results["summary"]["exceptions"] += 1
        
        # Store audit results
        self.store_audit_results(audit_results)
        
        # Generate audit report
        self.generate_audit_report(audit_results)
        
        return audit_results

    def test_access_certification(self, control):
        """Test access certification control"""
        
        test_result = {
            "status": "PASS",
            "findings": [],
            "evidence": {}
        }
        
        try:
            # Check certification completion rates
            certification_table = self.dynamodb.Table('CertificationTasks')
            
            # Get certifications from last quarter
            three_months_ago = datetime.now() - timedelta(days=90)
            
            response = certification_table.scan(
                FilterExpression='created_date >= :date',
                ExpressionAttributeValues={
                    ':date': three_months_ago.isoformat()
                }
            )
            
            total_certifications = response['Count']
            completed_certifications = 0
            overdue_certifications = 0
            
            current_time = datetime.now()
            
            for item in response['Items']:
                if item.get('status') == 'COMPLETED':
                    completed_certifications += 1
                elif item.get('status') == 'PENDING':
                    deadline = datetime.fromisoformat(item.get('deadline', ''))
                    if deadline < current_time:
                        overdue_certifications += 1
            
            # Calculate completion rate
            completion_rate = (completed_certifications / total_certifications * 100) if total_certifications > 0 else 0
            
            # Test criteria
            if completion_rate < 95:  # 95% completion rate required
                test_result["status"] = "FAIL"
                test_result["findings"].append({
                    "finding_id": f"{control['control_id']}-001",
                    "severity": "HIGH",
                    "description": f"Access certification completion rate below threshold: {completion_rate:.1f}%",
                    "affected_items": [f"Target: 95%, Actual: {completion_rate:.1f}%"],
                    "recommendation": "Improve certification process and follow-up procedures"
                })
            
            if overdue_certifications > 0:
                test_result["status"] = "FAIL"
                test_result["findings"].append({
                    "finding_id": f"{control['control_id']}-002",
                    "severity": "MEDIUM",
                    "description": f"Overdue certifications: {overdue_certifications}",
                    "affected_items": [f"{overdue_certifications} overdue items"],
                    "recommendation": "Escalate overdue certifications and implement automated reminders"
                })
            
            # Evidence
            test_result["evidence"] = {
                "total_certifications": total_certifications,
                "completed_certifications": completed_certifications,
                "completion_rate": completion_rate,
                "overdue_certifications": overdue_certifications
            }
            
        except Exception as e:
            test_result["status"] = "ERROR"
            test_result["error"] = str(e)
        
        return test_result

    def execute_athena_query(self, query):
        """Execute Athena query and return results"""
        
        try:
            # Start query execution
            response = self.athena_client.start_query_execution(
                QueryString=query,
                ResultConfiguration={
                    'OutputLocation': f's3://privilege-analytics-{boto3.Session().region_name}/athena-results/'
                }
            )
            
            query_execution_id = response['QueryExecutionId']
            
            # Wait for query completion
            import time
            while True:
                response = self.athena_client.get_query_execution(
                    QueryExecutionId=query_execution_id
                )
                
                status = response['QueryExecution']['Status']['State']
                
                if status in ['SUCCEEDED', 'FAILED', 'CANCELLED']:
                    break
                
                time.sleep(2)
            
            if status == 'SUCCEEDED':
                # Get query results
                results = self.athena_client.get_query_results(
                    QueryExecutionId=query_execution_id
                )
                
                # Parse results
                columns = [col['Name'] for col in results['ResultSet']['ResultSetMetadata']['ColumnInfo']]
                rows = []
                
                for row in results['ResultSet']['Rows'][1:]:  # Skip header row
                    row_data = {}
                    for i, cell in enumerate(row['Data']):
                        row_data[columns[i]] = cell.get('VarCharValue', '')
                    rows.append(row_data)
                
                return rows
            else:
                raise Exception(f"Query failed with status: {status}")
                
        except Exception as e:
            print(f"Error executing Athena query: {str(e)}")
            return []
    
    def store_audit_results(self, audit_results):
        """Store audit results in S3"""
        
        bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
        key = f'audit-results/{datetime.now().strftime("%Y/%m")}/audit-{audit_results["audit_id"]}.json'
        
        try:
            self.s3_client.put_object(
                Bucket=bucket_name,
                Key=key,
                Body=json.dumps(audit_results, indent=2, default=str),
                ContentType='application/json'
            )
            
            print(f"Audit results stored: s3://{bucket_name}/{key}")
            
        except Exception as e:
            print(f"Error storing audit results: {str(e)}")
    
    def generate_audit_report(self, audit_results):
        """Generate comprehensive audit report"""
        
        report = {
            "executive_summary": self.generate_executive_summary(audit_results),
            "detailed_findings": audit_results["findings"],
            "control_test_results": audit_results["controls_tested"],
            "recommendations": self.generate_recommendations(audit_results),
            "management_response": "Pending"
        }
        
        # Store report
        bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
        key = f'audit-reports/{datetime.now().strftime("%Y/%m")}/report-{audit_results["audit_id"]}.json'
        
        self.s3_client.put_object(
            Bucket=bucket_name,
            Key=key,
            Body=json.dumps(report, indent=2, default=str),
            ContentType='application/json'
        )
        
        return report
    
    def generate_executive_summary(self, audit_results):
        """Generate executive summary"""
        
        summary = audit_results["summary"]
        
        return {
            "audit_period": audit_results["audit_date"],
            "scope": audit_results["audit_scope"],
            "overall_rating": self.calculate_overall_rating(summary),
            "key_metrics": {
                "total_controls_tested": summary["total_controls"],
                "controls_passed": summary["passed"],
                "controls_failed": summary["failed"],
                "pass_rate": (summary["passed"] / summary["total_controls"] * 100) if summary["total_controls"] > 0 else 0
            },
            "critical_findings": len([f for f in audit_results["findings"] if f.get("severity") == "CRITICAL"]),
            "high_findings": len([f for f in audit_results["findings"] if f.get("severity") == "HIGH"])
        }
    
    def calculate_overall_rating(self, summary):
        """Calculate overall audit rating"""
        
        if summary["total_controls"] == 0:
            return "INSUFFICIENT_DATA"
        
        pass_rate = summary["passed"] / summary["total_controls"] * 100
        
        if pass_rate >= 95:
            return "SATISFACTORY"
        elif pass_rate >= 85:
            return "NEEDS_IMPROVEMENT"
        else:
            return "UNSATISFACTORY"

def lambda_handler(event, context):
    """Lambda handler for audit execution"""
    
    audit_framework = AuditFramework()
    
    # Get audit scope from event
    audit_scope = event.get('audit_scope', 'full')
    
    # Execute audit
    results = audit_framework.execute_audit_cycle(audit_scope)
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Audit completed successfully',
            'audit_id': results['audit_id'],
            'controls_tested': results['summary']['total_controls'],
            'findings': len(results['findings']),
            'pass_rate': (results['summary']['passed'] / results['summary']['total_controls'] * 100) if results['summary']['total_controls'] > 0 else 0
        })
    }

import boto3
import json
from datetime import datetime

class ContinuousAuditMonitor:
    def __init__(self):
        self.config_client = boto3.client('config')
        self.cloudwatch = boto3.client('cloudwatch')
        self.sns = boto3.client('sns')
    
    def monitor_compliance_rules(self):
        """Monitor AWS Config compliance rules"""
        
        monitoring_results = {
            "timestamp": datetime.now().isoformat(),
            "compliance_summary": {},
            "non_compliant_resources": [],
            "alerts_sent": []
        }
        
        # Get compliance summary
        response = self.config_client.get_compliance_summary_by_config_rule()
        
        for rule_summary in response['ComplianceSummary']:
            rule_name = rule_summary['ConfigRuleName']
            compliance_summary = rule_summary['ComplianceSummary']
            
            monitoring_results["compliance_summary"][rule_name] = {
                "compliant": compliance_summary.get('CompliantResourceCount', {}).get('CappedCount', 0),
                "non_compliant": compliance_summary.get('NonCompliantResourceCount', {}).get('CappedCount', 0),
                "insufficient_data": compliance_summary.get('InsufficientDataResourceCount', {}).get('CappedCount', 0)
            }
            
            # Check for non-compliant resources
            if compliance_summary.get('NonCompliantResourceCount', {}).get('CappedCount', 0) > 0:
                non_compliant = self.get_non_compliant_resources(rule_name)
                monitoring_results["non_compliant_resources"].extend(non_compliant)
        
        # Send alerts for critical non-compliance
        critical_violations = [
            resource for resource in monitoring_results["non_compliant_resources"]
            if resource.get("severity") == "CRITICAL"
        ]
        
        if critical_violations:
            self.send_compliance_alert(critical_violations)
            monitoring_results["alerts_sent"].append("CRITICAL_COMPLIANCE_VIOLATIONS")
        
        return monitoring_results
    
    def get_non_compliant_resources(self, rule_name):
        """Get details of non-compliant resources"""
        
        try:
            response = self.config_client.get_compliance_details_by_config_rule(
                ConfigRuleName=rule_name,
                ComplianceTypes=['NON_COMPLIANT']
            )
            
            non_compliant_resources = []
            
            for result in response['EvaluationResults']:
                resource = {
                    "rule_name": rule_name,
                    "resource_type": result['EvaluationResultIdentifier']['EvaluationResultQualifier']['ResourceType'],
                    "resource_id": result['EvaluationResultIdentifier']['EvaluationResultQualifier']['ResourceId'],
                    "compliance_type": result['ComplianceType'],
                    "result_recorded_time": result['ResultRecordedTime'].isoformat(),
                    "severity": self.determine_severity(rule_name)
                }
                
                non_compliant_resources.append(resource)
            
            return non_compliant_resources
            
        except Exception as e:
            print(f"Error getting non-compliant resources for {rule_name}: {str(e)}")
            return []
    
    def determine_severity(self, rule_name):
        """Determine severity based on rule name"""
        
        critical_rules = [
            'root-mfa-enabled',
            'iam-password-policy',
            'cloudtrail-enabled'
        ]
        
        high_rules = [
            'iam-user-mfa-enabled',
            'access-keys-rotated'
        ]
        
        if any(critical in rule_name.lower() for critical in critical_rules):
            return "CRITICAL"
        elif any(high in rule_name.lower() for high in high_rules):
            return "HIGH"
        else:
            return "MEDIUM"
    
    def send_compliance_alert(self, violations):
        """Send alert for compliance violations"""
        
        message = f"""
        CRITICAL COMPLIANCE VIOLATIONS DETECTED
        
        Timestamp: {datetime.now().isoformat()}
        Total Violations: {len(violations)}
        
        Critical Violations:
        """
        
        for violation in violations[:10]:  # Limit to first 10
            message += f"""
        - Rule: {violation['rule_name']}
          Resource: {violation['resource_id']}
          Type: {violation['resource_type']}
        """
        
        try:
            self.sns.publish(
                TopicArn='arn:aws:sns:region:account:compliance-alerts',
                Subject='CRITICAL: Compliance Violations Detected',
                Message=message
            )
        except Exception as e:
            print(f"Error sending compliance alert: {str(e)}")

def lambda_handler(event, context):

    monitor = ContinuousAuditMonitor()

    return {

            'message': 'Compliance monitoring completed',

3.1 Automated Report Generator

import boto3
import json
from datetime import datetime, timedelta
from jinja2 import Template

class AuditReportGenerator:
    def __init__(self):
        self.s3_client = boto3.client('s3')
        self.ses_client = boto3.client('ses')
    
    def generate_monthly_audit_report(self):
        """Generate comprehensive monthly audit report"""
        
        report_data = {
            "report_period": datetime.now().strftime("%Y-%m"),
            "generated_date": datetime.now().isoformat(),
            "executive_summary": {},
            "control_effectiveness": {},
            "findings_summary": {},
            "trend_analysis": {},
            "recommendations": []
        }
        
        # Collect audit data from S3
        audit_data = self.collect_monthly_audit_data()
        
        # Generate report sections
        report_data["executive_summary"] = self.generate_executive_summary(audit_data)
        report_data["control_effectiveness"] = self.analyze_control_effectiveness(audit_data)
        report_data["findings_summary"] = self.summarize_findings(audit_data)
        report_data["trend_analysis"] = self.analyze_trends(audit_data)
        report_data["recommendations"] = self.generate_recommendations(audit_data)
        
        # Generate HTML report
        html_report = self.generate_html_report(report_data)
        
        # Store report
        self.store_report(report_data, html_report)
        
        # Send report to stakeholders
        self.distribute_report(html_report)
        
        return report_data
    
    def collect_monthly_audit_data(self):
        """Collect audit data from the past month"""
        
        bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
        current_month = datetime.now().strftime("%Y/%m")
        
        audit_data = {
            "audit_results": [],
            "compliance_data": [],
            "findings": []
        }
        
        try:
            # List audit results for current month
            response = self.s3_client.list_objects_v2(
                Bucket=bucket_name,
                Prefix=f'audit-results/{current_month}/'
            )
            
            for obj in response.get('Contents', []):
                # Get audit result
                result = self.s3_client.get_object(
                    Bucket=bucket_name,
                    Key=obj['Key']
                )
                
                audit_result = json.loads(result['Body'].read())
                audit_data["audit_results"].append(audit_result)
                audit_data["findings"].extend(audit_result.get("findings", []))
            
        except Exception as e:
            print(f"Error collecting audit data: {str(e)}")
        
        return audit_data

    def store_report(self, report_data, html_report):
        """Store audit report in S3"""
        
        bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
        
        # Store JSON data
        json_key = f'audit-reports/{report_data["report_period"]}/monthly-report.json'
        self.s3_client.put_object(
            Bucket=bucket_name,
            Key=json_key,
            Body=json.dumps(report_data, indent=2, default=str),
            ContentType='application/json'
        )
        
        # Store HTML report
        html_key = f'audit-reports/{report_data["report_period"]}/monthly-report.html'
        self.s3_client.put_object(
            Bucket=bucket_name,
            Key=html_key,
            Body=html_report,
            ContentType='text/html'
        )
        
        print(f"Report stored: s3://{bucket_name}/{html_key}")

                    },
                    'Body': {
                        'Html': {'Data': html_report}
                    }
                }
            )
            
            print(f"Report distributed to {len(recipients)} recipients")
            
        except Exception as e:
            print(f"Error distributing report: {str(e)}")

def lambda_handler(event, context):
