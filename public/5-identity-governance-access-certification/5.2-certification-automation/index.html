Certification Automation

Bước 2: Access Review Generator

2.1 Lambda Function cho Data Collection

import boto3
import json
from datetime import datetime, timedelta

class AccessReviewGenerator:
    def __init__(self):
        self.sso_client = boto3.client('sso-admin')
        self.identity_client = boto3.client('identitystore')
        self.org_client = boto3.client('organizations')
        self.s3_client = boto3.client('s3')
        
    def generate_access_review(self):
        """Generate comprehensive access review data"""
        
        # Get all accounts
        accounts = self.get_all_accounts()
        
        # Get all permission sets
        permission_sets = self.get_all_permission_sets()
        
        # Get all assignments
        assignments = self.get_all_assignments(accounts, permission_sets)
        
        # Generate review data
        review_data = {
            'review_id': f"review_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
            'generated_date': datetime.now().isoformat(),
            'accounts': accounts,
            'permission_sets': permission_sets,
            'assignments': assignments,
            'review_deadline': (datetime.now() + timedelta(days=30)).isoformat()
        }
        
        # Store in S3
        self.store_review_data(review_data)
        
        return review_data
    
    def get_all_accounts(self):
        """Get all AWS accounts in organization"""
        accounts = []
        paginator = self.org_client.get_paginator('list_accounts')
        
        for page in paginator.paginate():
            for account in page['Accounts']:
                accounts.append({
                    'account_id': account['Id'],
                    'account_name': account['Name'],
                    'email': account['Email'],
                    'status': account['Status']
                })
        
        return accounts
    
    def get_all_permission_sets(self):
        """Get all permission sets"""
        instance_arn = self.get_instance_arn()
        permission_sets = []
        
        paginator = self.sso_client.get_paginator('list_permission_sets')
        
        for page in paginator.paginate(InstanceArn=instance_arn):
            for ps_arn in page['PermissionSets']:
                ps_details = self.sso_client.describe_permission_set(
                    InstanceArn=instance_arn,
                    PermissionSetArn=ps_arn
                )
                
                permission_sets.append({
                    'arn': ps_arn,
                    'name': ps_details['PermissionSet']['Name'],
                    'description': ps_details['PermissionSet'].get('Description', ''),
                    'session_duration': ps_details['PermissionSet'].get('SessionDuration', '')
                })
        
        return permission_sets
    
    def get_all_assignments(self, accounts, permission_sets):
        """Get all account assignments"""
        instance_arn = self.get_instance_arn()
        assignments = []
        
        for account in accounts:
            for ps in permission_sets:
                try:
                    paginator = self.sso_client.get_paginator('list_account_assignments')
                    
                    for page in paginator.paginate(
                        InstanceArn=instance_arn,
                        AccountId=account['account_id'],
                        PermissionSetArn=ps['arn']
                    ):
                        for assignment in page['AccountAssignments']:
                            # Get principal details
                            principal_details = self.get_principal_details(
                                assignment['PrincipalId'],
                                assignment['PrincipalType']
                            )
                            
                            assignments.append({
                                'account_id': account['account_id'],
                                'account_name': account['account_name'],
                                'permission_set_name': ps['name'],
                                'permission_set_arn': ps['arn'],
                                'principal_id': assignment['PrincipalId'],
                                'principal_type': assignment['PrincipalType'],
                                'principal_name': principal_details.get('name', ''),
                                'principal_email': principal_details.get('email', ''),
                                'manager_email': principal_details.get('manager_email', '')
                            })
                            
                except Exception as e:
                    print(f"Error getting assignments for {account['account_id']}: {str(e)}")
                    continue
        
        return assignments
    
    def get_principal_details(self, principal_id, principal_type):
        """Get details about user or group"""
        identity_store_id = self.get_identity_store_id()
        
        try:
            if principal_type == 'USER':
                response = self.identity_client.describe_user(
                    IdentityStoreId=identity_store_id,
                    UserId=principal_id
                )
                user = response['User']
                return {
                    'name': user.get('DisplayName', user.get('UserName', '')),
                    'email': user.get('Emails', [{}])[0].get('Value', ''),
                    'manager_email': self.get_manager_email(principal_id)
                }
            elif principal_type == 'GROUP':
                response = self.identity_client.describe_group(
                    IdentityStoreId=identity_store_id,
                    GroupId=principal_id
                )
                group = response['Group']
                return {
                    'name': group.get('DisplayName', ''),
                    'email': '',
                    'manager_email': ''
                }
        except Exception as e:
            print(f"Error getting principal details: {str(e)}")
            return {}
    
    def get_manager_email(self, user_id):
        """Get manager email from user attributes"""
        # Implementation depends on your identity store setup
        # This is a placeholder
        return "manager@company.com"
    
    def store_review_data(self, review_data):
        """Store review data in S3"""
        bucket_name = 'access-certification-data'
        key = f"reviews/{review_data['review_id']}.json"
        
        self.s3_client.put_object(
            Bucket=bucket_name,
            Key=key,
            Body=json.dumps(review_data, indent=2),
            ContentType='application/json'
        )
        
        print(f"Review data stored: s3://{bucket_name}/{key}")
    
    def get_instance_arn(self):
        """Get SSO instance ARN"""
        response = self.sso_client.list_instances()
        return response['Instances'][0]['InstanceArn']
    
    def get_identity_store_id(self):
        """Get identity store ID"""
        response = self.sso_client.list_instances()
        return response['Instances'][0]['IdentityStoreId']

def lambda_handler(event, context):
    generator = AccessReviewGenerator()
    review_data = generator.generate_access_review()
    
    # Trigger notification workflow
    trigger_certification_workflow(review_data)
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Access review generated successfully',
            'review_id': review_data['review_id']
        })
    }

def trigger_certification_workflow(review_data):
    """Trigger Step Functions workflow for certification"""
    stepfunctions = boto3.client('stepfunctions')
    
    stepfunctions.start_execution(
        stateMachineArn='arn:aws:states:region:account:stateMachine:CertificationWorkflow',
        input=json.dumps(review_data)
    )

3.2 Notification Lambda Function

import boto3
import json
from jinja2 import Template

class CertificationNotifier:
    def __init__(self):
        self.ses_client = boto3.client('ses')
        self.dynamodb = boto3.resource('dynamodb')
        self.table = self.dynamodb.Table('CertificationTasks')
    
    def send_certification_notifications(self, review_data):
        """Send certification emails to managers"""
        
        # Group assignments by manager
        manager_assignments = self.group_by_manager(review_data['assignments'])
        
        for manager_email, assignments in manager_assignments.items():
            if manager_email:
                self.send_manager_notification(manager_email, assignments, review_data)
        
        return {'notifications_sent': len(manager_assignments)}
    
    def group_by_manager(self, assignments):
        """Group assignments by manager email"""
        manager_groups = {}
        
        for assignment in assignments:
            manager_email = assignment.get('manager_email', 'unknown@company.com')
            
            if manager_email not in manager_groups:
                manager_groups[manager_email] = []
            
            manager_groups[manager_email].append(assignment)
        
        return manager_groups
    
    def send_manager_notification(self, manager_email, assignments, review_data):
        """Send certification email to manager"""
        
        # Create certification tasks
        task_id = f"{review_data['review_id']}_{manager_email.replace('@', '_at_')}"
        
        # Store task in DynamoDB
        self.table.put_item(
            Item={
                'task_id': task_id,
                'review_id': review_data['review_id'],
                'manager_email': manager_email,
                'assignments': assignments,
                'status': 'PENDING',
                'created_date': review_data['generated_date'],
                'deadline': review_data['review_deadline']
            }
        )
        
        # Generate email content
        email_content = self.generate_email_content(manager_email, assignments, task_id)
        
        # Send email
        self.ses_client.send_email(
            Source='noreply@company.com',
            Destination={'ToAddresses': [manager_email]},
            Message={
                'Subject': {'Data': 'Access Certification Required - Action Needed'},
                'Body': {
                    'Html': {'Data': email_content}
                }
            }
        )
        
        print(f"Certification email sent to {manager_email}")
    
    def generate_email_content(self, manager_email, assignments, task_id):
        """Generate HTML email content"""
        
        template = Template("""
        <html>
        <head>
            <style>
                body { font-family: Arial, sans-serif; }
                .header { background-color: #f0f0f0; padding: 20px; }
                .content { padding: 20px; }
                .assignment { border: 1px solid #ddd; margin: 10px 0; padding: 10px; }
                .approve { background-color: #4CAF50; color: white; padding: 10px 20px; text-decoration: none; }
                .deny { background-color: #f44336; color: white; padding: 10px 20px; text-decoration: none; }
            </style>
        </head>
        <body>
            <div class="header">
                <h2>Access Certification Review Required</h2>
            </div>
            <div class="content">
                <p>Dear Manager,</p>
                <p>Please review and certify the following access assignments for your team members:</p>
                
                {% for assignment in assignments %}
                <div class="assignment">
                    <h4>{{ assignment.principal_name }} ({{ assignment.principal_email }})</h4>
                    <p><strong>Account:</strong> {{ assignment.account_name }} ({{ assignment.account_id }})</p>
                    <p><strong>Permission Set:</strong> {{ assignment.permission_set_name }}</p>
                    <p>
                        <a href="https://certification.company.com/approve/{{ task_id }}/{{ loop.index0 }}" class="approve">APPROVE</a>
                        <a href="https://certification.company.com/deny/{{ task_id }}/{{ loop.index0 }}" class="deny">DENY</a>
                    </p>
                </div>
                {% endfor %}
                
                <p><strong>Deadline:</strong> Please complete this review within 30 days.</p>
                <p>If you have any questions, please contact the Security team.</p>
            </div>
        </body>
        </html>
        """)
        
        return template.render(assignments=assignments, task_id=task_id)

def lambda_handler(event, context):
    notifier = CertificationNotifier()
    result = notifier.send_certification_notifications(event)
    
    return {
        'statusCode': 200,
        'body': json.dumps(result)
    }

Bước 4: Web Interface cho Certification

4.1 API Gateway và Lambda

import boto3
import json
from datetime import datetime

class CertificationAPI:
    def __init__(self):
        self.dynamodb = boto3.resource('dynamodb')
        self.table = self.dynamodb.Table('CertificationTasks')
        self.sso_client = boto3.client('sso-admin')
    
    def handle_certification_response(self, event, context):
        """Handle certification approval/denial"""
        
        path_params = event.get('pathParameters', {})
        action = path_params.get('action')  # approve or deny
        task_id = path_params.get('task_id')
        assignment_index = int(path_params.get('assignment_index', 0))
        
        # Get task from DynamoDB
        response = self.table.get_item(Key={'task_id': task_id})
        
        if 'Item' not in response:
            return {
                'statusCode': 404,
                'body': json.dumps({'error': 'Task not found'})
            }
        
        task = response['Item']
        assignments = task['assignments']
        
        if assignment_index >= len(assignments):
            return {
                'statusCode': 400,
                'body': json.dumps({'error': 'Invalid assignment index'})
            }
        
        assignment = assignments[assignment_index]
        
        # Update assignment status
        assignment['certification_status'] = action.upper()
        assignment['certification_date'] = datetime.now().isoformat()
        
        # Update task in DynamoDB
        self.table.update_item(
            Key={'task_id': task_id},
            UpdateExpression='SET assignments = :assignments',
            ExpressionAttributeValues={':assignments': assignments}
        )
        
        # If denied, trigger remediation
        if action == 'deny':
            self.trigger_remediation(assignment)
        
        return {
            'statusCode': 200,
            'headers': {'Content-Type': 'text/html'},
            'body': self.generate_response_page(action, assignment)
        }
    
    def trigger_remediation(self, assignment):
        """Trigger access removal for denied assignments"""
        
        instance_arn = self.get_instance_arn()
        
        try:
            # Remove account assignment
            self.sso_client.delete_account_assignment(
                InstanceArn=instance_arn,
                TargetId=assignment['account_id'],
                TargetType='AWS_ACCOUNT',
                PermissionSetArn=assignment['permission_set_arn'],
                PrincipalType=assignment['principal_type'],
                PrincipalId=assignment['principal_id']
            )
            
            print(f"Removed access for {assignment['principal_name']}")
            
        except Exception as e:
            print(f"Error removing access: {str(e)}")
    
    def generate_response_page(self, action, assignment):
        """Generate HTML response page"""
        
        if action == 'approve':
            message = f"Access APPROVED for {assignment['principal_name']}"
            color = "green"
        else:
            message = f"Access DENIED for {assignment['principal_name']} - Remediation initiated"
            color = "red"
        
        return f"""
        <html>
        <body style="font-family: Arial, sans-serif; text-align: center; padding: 50px;">
            <h2 style="color: {color};">{message}</h2>
            <p>Thank you for completing the access certification.</p>
            <p>This action has been logged and processed automatically.</p>
        </body>
        </html>
        """
    
    def get_instance_arn(self):
        """Get SSO instance ARN"""
        response = self.sso_client.list_instances()
        return response['Instances'][0]['InstanceArn']

def lambda_handler(event, context):
    api = CertificationAPI()
    return api.handle_certification_response(event, context)

Bước 5: Deployment Script

5.1 Complete Deployment

#!/bin/bash

# Deploy infrastructure
aws cloudformation deploy \
  --template-file certification-automation.yaml \
  --stack-name access-certification \
  --capabilities CAPABILITY_IAM

# Create S3 bucket for data
aws s3 mb s3://access-certification-data

# Create DynamoDB table
aws dynamodb create-table \
  --table-name CertificationTasks \
  --attribute-definitions AttributeName=task_id,AttributeType=S \
  --key-schema AttributeName=task_id,KeyType=HASH \
  --billing-mode PAY_PER_REQUEST

# Deploy Lambda functions
zip -r access-review-generator.zip access_review_generator.py
aws lambda create-function \
  --function-name AccessReviewGenerator \
  --runtime python3.9 \
  --role arn:aws:iam::ACCOUNT:role/LambdaExecutionRole \
  --handler access_review_generator.lambda_handler \
  --zip-file fileb://access-review-generator.zip

echo "Certification automation deployed successfully!"
