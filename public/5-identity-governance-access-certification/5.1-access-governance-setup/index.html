5.1 Access Governance Setup

Access Governance Setup

Mục tiêu

Thiết lập nền tảng quản lý truy cập tập trung với AWS IAM Identity Center và Organizations.

Kiến trúc

graph LR
    A[Management Account] --> B[AWS Organizations]
    B --> C[Member Accounts]
    C --> D[IAM Identity Center]
    D --> E[Permission Sets]
    E --> F[User Assignments]

Bước 1: Thiết lập AWS Organizations

1.1 Tạo Organization

aws organizations create-organization --feature-set ALL

1.2 Tạo Organizational Units

{
  "Name": "Security",
  "ParentId": "r-xxxx"
}

aws organizations create-organizational-unit \
  --parent-id r-xxxx \
  --name "Security"

Bước 2: Cấu hình IAM Identity Center

2.1 Enable IAM Identity Center

aws sso-admin create-instance \
  --name "IdentityGovernance" \
  --description "Identity Governance Instance"

Bước 3: Thiết lập Identity Store

3.1 Cấu hình External Identity Provider

import boto3

def setup_external_idp():
    client = boto3.client('identitystore')
    
    # Configure SAML IdP
    response = client.create_external_id_provider(
        IdentityStoreId='d-xxxxxxxxxx',
        ExternalIdProvider={
            'Type': 'SAML',
            'Configuration': {
                'MetadataDocument': 'metadata_content',
                'LoginUrl': 'https://idp.company.com/login',
                'LogoutUrl': 'https://idp.company.com/logout'
            }
        }
    )
    return response

3.2 Tạo Groups và Users

def create_governance_groups():
    client = boto3.client('identitystore')
    
    groups = [
        {
            'DisplayName': 'SecurityAdmins',
            'Description': 'Security administrators with full access'
        },
        {
            'DisplayName': 'Auditors',
            'Description': 'Security auditors with read-only access'
        },
        {
            'DisplayName': 'ComplianceOfficers',
            'Description': 'Compliance officers for certification'
        }
    ]
    
    for group in groups:
        response = client.create_group(
            IdentityStoreId='d-xxxxxxxxxx',
            DisplayName=group['DisplayName'],
            Description=group['Description']
        )
        print(f"Created group: {group['DisplayName']}")

Bước 4: Automation Scripts

4.1 Permission Set Deployment

import boto3
import json

class PermissionSetManager:
    def __init__(self):
        self.sso_client = boto3.client('sso-admin')
        self.instance_arn = self.get_instance_arn()
    
    def get_instance_arn(self):
        response = self.sso_client.list_instances()
        return response['Instances'][0]['InstanceArn']
    
    def create_permission_set(self, name, description, policies):
        response = self.sso_client.create_permission_set(
            Name=name,
            Description=description,
            InstanceArn=self.instance_arn,
            SessionDuration='PT8H'
        )
        
        permission_set_arn = response['PermissionSet']['PermissionSetArn']
        
        # Attach managed policies
        for policy in policies.get('managed', []):
            self.sso_client.attach_managed_policy_to_permission_set(
                InstanceArn=self.instance_arn,
                PermissionSetArn=permission_set_arn,
                ManagedPolicyArn=policy
            )
        
        # Attach inline policy
        if 'inline' in policies:
            self.sso_client.put_inline_policy_to_permission_set(
                InstanceArn=self.instance_arn,
                PermissionSetArn=permission_set_arn,
                InlinePolicy=json.dumps(policies['inline'])
            )
        
        return permission_set_arn

# Usage
manager = PermissionSetManager()
governance_policies = {
    'managed': [
        'arn:aws:iam::aws:policy/SecurityAudit'
    ],
    'inline': {
        "Version": "2012-10-17",
        "Statement": [
            {
                "Effect": "Allow",
                "Action": [
                    "access-analyzer:*",
                    "config:*",
                    "cloudtrail:*"
                ],
                "Resource": "*"
            }
        ]
    }
}

permission_set_arn = manager.create_permission_set(
    "GovernanceAnalyst",
    "Identity governance analyst role",
    governance_policies
)

5.1 Test Script

def validate_governance_setup():
    sso_client = boto3.client('sso-admin')
    org_client = boto3.client('organizations')
    
    # Check Organizations setup
    org_response = org_client.describe_organization()
    print(f"Organization ID: {org_response['Organization']['Id']}")
    
    # Check IAM Identity Center
    instances = sso_client.list_instances()
    print(f"SSO Instances: {len(instances['Instances'])}")
    
    # List Permission Sets
    instance_arn = instances['Instances'][0]['InstanceArn']
    permission_sets = sso_client.list_permission_sets(InstanceArn=instance_arn)
    print(f"Permission Sets: {len(permission_sets['PermissionSets'])}")
    
    return True

validate_governance_setup()

Troubleshooting

Lỗi thường gặp:

  1. Organizations not enabled: Đảm bảo account có quyền tạo organization
  2. IAM Identity Center region: Chỉ có thể enable ở một region
  3. Permission denied: Kiểm tra IAM permissions cho service-linked roles
