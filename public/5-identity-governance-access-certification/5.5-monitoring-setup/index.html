import boto3
import json
import os
from datetime import datetime, timedelta

class IdentityGovernanceMonitor:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.logs_client = boto3.client('logs')
        self.iam_client = boto3.client('iam')
        self.sso_client = boto3.client('sso-admin')
        self.sns_client = boto3.client('sns')
        
        self.log_group = os.environ.get('LOG_GROUP_NAME', '/aws/identity-governance/events')
        self.sns_topic = os.environ.get('SNS_TOPIC_ARN')
    
    def monitor_identity_events(self):
        """Monitor and analyze identity-related events"""
        
        monitoring_results = {
            'timestamp': datetime.now().isoformat(),
            'metrics_collected': [],
            'anomalies_detected': [],
            'alerts_sent': []
        }
        
        # Collect various metrics
        metrics = [
            self.collect_login_metrics(),
            self.collect_permission_changes(),
            self.collect_access_patterns(),
            self.collect_certification_metrics(),
            self.collect_risk_metrics()
        ]
        
        for metric_data in metrics:
            if metric_data:
                self.send_metrics_to_cloudwatch(metric_data)
                monitoring_results['metrics_collected'].append(metric_data['metric_name'])
        
        # Detect anomalies
        anomalies = self.detect_anomalies()
        monitoring_results['anomalies_detected'] = anomalies
        
        # Send alerts for critical issues
        for anomaly in anomalies:
            if anomaly['severity'] == 'CRITICAL':
                self.send_alert(anomaly)
                monitoring_results['alerts_sent'].append(anomaly['type'])
        
        return monitoring_results
    
    def collect_login_metrics(self):
        """Collect login-related metrics"""
        
        end_time = datetime.now()
        start_time = end_time - timedelta(hours=1)
        
        try:
            # Query CloudTrail logs for login events
            response = self.logs_client.start_query(
                logGroupName='/aws/cloudtrail',
                startTime=int(start_time.timestamp()),
                endTime=int(end_time.timestamp()),
                queryString='''
                    fields @timestamp, sourceIPAddress, userIdentity.userName, errorCode
                    | filter eventName = "ConsoleLogin"
                    | stats count() by errorCode
                '''
            )
            
            query_id = response['queryId']
            
            # Wait for query completion and get results
            import time
            time.sleep(5)
            
            results = self.logs_client.get_query_results(queryId=query_id)
            
            successful_logins = 0
            failed_logins = 0
            
            for result in results.get('results', []):
                if len(result) >= 2:
                    error_code = result[0]['value']
                    count = int(result[1]['value'])
                    
                    if error_code == 'Success':
                        successful_logins = count
                    else:
                        failed_logins += count
            
            return {
                'metric_name': 'LoginMetrics',
                'metrics': [
                    {
                        'MetricName': 'SuccessfulLogins',
                        'Value': successful_logins,
                        'Unit': 'Count'
                    },
                    {
                        'MetricName': 'FailedLogins',
                        'Value': failed_logins,
                        'Unit': 'Count'
                    }
                ]
            }
            
        except Exception as e:
            print(f"Error collecting login metrics: {str(e)}")
            return None
    
    def collect_permission_changes(self):
        """Collect metrics on permission changes"""
        
        end_time = datetime.now()
        start_time = end_time - timedelta(hours=1)
        
        try:
            # Query for IAM changes
            response = self.logs_client.start_query(
                logGroupName='/aws/cloudtrail',
                startTime=int(start_time.timestamp()),
                endTime=int(end_time.timestamp()),
                queryString='''
                    fields @timestamp, eventName, userIdentity.userName
                    | filter eventName like /^(AttachUserPolicy|DetachUserPolicy|CreateRole|DeleteRole|PutUserPolicy|DeleteUserPolicy)$/
                    | stats count() by eventName
                '''
            )
            
            query_id = response['queryId']
            
            import time
            time.sleep(5)
            
            results = self.logs_client.get_query_results(queryId=query_id)
            
            permission_changes = sum(
                int(result[1]['value']) for result in results.get('results', [])
                if len(result) >= 2
            )
            
            return {
                'metric_name': 'PermissionChanges',
                'metrics': [
                    {
                        'MetricName': 'PermissionChanges',
                        'Value': permission_changes,
                        'Unit': 'Count'
                    }
                ]
            }
            
        except Exception as e:
            print(f"Error collecting permission change metrics: {str(e)}")
            return None
    
    def collect_access_patterns(self):
        """Collect access pattern metrics"""
        
        try:
            # Get current hour
            current_hour = datetime.now().hour
            
            # Determine if this is off-hours access
            is_off_hours = current_hour < 6 or current_hour > 22
            
            # Query recent access events
            end_time = datetime.now()
            start_time = end_time - timedelta(hours=1)
            
            response = self.logs_client.start_query(
                logGroupName='/aws/cloudtrail',
                startTime=int(start_time.timestamp()),
                endTime=int(end_time.timestamp()),
                queryString='''
                    fields @timestamp, sourceIPAddress, userIdentity.userName
                    | filter eventName = "AssumeRole"
                    | stats count() by sourceIPAddress
                '''
            )
            
            query_id = response['queryId']
            
            import time
            time.sleep(5)
            
            results = self.logs_client.get_query_results(queryId=query_id)
            
            external_access_count = 0
            total_access_count = 0
            
            for result in results.get('results', []):
                if len(result) >= 2:
                    source_ip = result[0]['value']
                    count = int(result[1]['value'])
                    total_access_count += count
                    
                    # Check if IP is external
                    if not self.is_internal_ip(source_ip):
                        external_access_count += count
            
            return {
                'metric_name': 'AccessPatterns',
                'metrics': [
                    {
                        'MetricName': 'ExternalAccess',
                        'Value': external_access_count,
                        'Unit': 'Count'
                    },
                    {
                        'MetricName': 'OffHoursAccess',
                        'Value': total_access_count if is_off_hours else 0,
                        'Unit': 'Count'
                    }
                ]
            }
            
        except Exception as e:
            print(f"Error collecting access pattern metrics: {str(e)}")
            return None

    def send_metrics_to_cloudwatch(self, metric_data):
        """Send metrics to CloudWatch"""
        
        try:
            self.cloudwatch.put_metric_data(
                Namespace='IdentityGovernance',
                MetricData=metric_data['metrics']
            )
            
            print(f"Sent {len(metric_data['metrics'])} metrics to CloudWatch")
            
        except Exception as e:
            print(f"Error sending metrics to CloudWatch: {str(e)}")
    
    def detect_anomalies(self):
        """Detect anomalies in identity governance"""
        
        anomalies = []
        
        # Check for unusual login patterns
        failed_login_anomaly = self.detect_failed_login_anomaly()
        if failed_login_anomaly:
            anomalies.append(failed_login_anomaly)
        
        # Check for privilege escalation
        privilege_escalation = self.detect_privilege_escalation()
        if privilege_escalation:
            anomalies.append(privilege_escalation)
        
        # Check for off-hours access
        off_hours_anomaly = self.detect_off_hours_anomaly()
        if off_hours_anomaly:
            anomalies.append(off_hours_anomaly)
        
        return anomalies
    
    def detect_failed_login_anomaly(self):
        """Detect anomalous failed login patterns"""
        
        try:
            # Get failed login count for last hour
            response = self.cloudwatch.get_metric_statistics(
                Namespace='IdentityGovernance',

                StartTime=datetime.now() - timedelta(hours=1),

                        'type': 'FAILED_LOGIN_SPIKE',
                        'severity': 'HIGH',
                        'description': f'Unusual spike in failed logins: {failed_logins}',
                        'timestamp': datetime.now().isoformat(),
                        'value': failed_logins
                    }

2.2 Automated Log Analysis

import boto3
import json
from datetime import datetime, timedelta

class LogAnalyzer:
    def __init__(self):
        self.logs_client = boto3.client('logs')
        self.cloudwatch = boto3.client('cloudwatch')
    
    def analyze_security_logs(self):
        """Analyze security logs for patterns and anomalies"""
        
        analysis_results = {
            'analysis_timestamp': datetime.now().isoformat(),
            'patterns_detected': [],
            'recommendations': []
        }
        
        # Run various analysis queries
        analyses = [
            self.analyze_failed_logins(),
            self.analyze_privilege_changes(),
            self.analyze_access_patterns(),
            self.analyze_user_behavior()
        ]
        
        for analysis in analyses:
            if analysis:
                analysis_results['patterns_detected'].extend(analysis.get('patterns', []))
                analysis_results['recommendations'].extend(analysis.get('recommendations', []))
        
        return analysis_results
    
    def analyze_failed_logins(self):
        """Analyze failed login patterns"""
        
        query = '''
        fields @timestamp, sourceIPAddress, userIdentity.userName, errorMessage
        | filter eventName = "ConsoleLogin" and errorCode != "Success"
        | stats count() by sourceIPAddress, userIdentity.userName
        | sort count desc
        '''
        
        results = self.run_insights_query(query, hours_back=24)
        
        patterns = []
        recommendations = []
        
        for result in results:
            if len(result) >= 3:
                source_ip = result[0]['value']
                username = result[1]['value']
                count = int(result[2]['value'])
                
                if count > 10:  # Threshold for suspicious activity
                    patterns.append({
                        'type': 'SUSPICIOUS_FAILED_LOGINS',
                        'source_ip': source_ip,
                        'username': username,
                        'count': count,
                        'severity': 'HIGH' if count > 20 else 'MEDIUM'
                    })
                    
                    recommendations.append({
                        'type': 'SECURITY_ACTION',
                        'description': f'Investigate failed login attempts from {source_ip} for user {username}',
                        'priority': 'HIGH' if count > 20 else 'MEDIUM'
                    })
        
        return {
            'patterns': patterns,
            'recommendations': recommendations
        }
    
    def analyze_privilege_changes(self):
        """Analyze privilege escalation patterns"""
        
        query = '''
        fields @timestamp, eventName, userIdentity.userName, sourceIPAddress
        | filter eventName in ["AttachUserPolicy", "CreateRole", "PutUserPolicy", "AttachRolePolicy"]
        | stats count() by userIdentity.userName, eventName
        | sort count desc
        '''
        
        results = self.run_insights_query(query, hours_back=24)
        
        patterns = []
        recommendations = []
        
        for result in results:
            if len(result) >= 3:
                username = result[0]['value']
                event_name = result[1]['value']
                count = int(result[2]['value'])
                
                if count > 5:  # Threshold for unusual privilege changes
                    patterns.append({
                        'type': 'PRIVILEGE_ESCALATION_PATTERN',
                        'username': username,
                        'event_name': event_name,
                        'count': count,
                        'severity': 'HIGH'
                    })
                    
                    recommendations.append({
                        'type': 'PRIVILEGE_REVIEW',
                        'description': f'Review privilege changes by {username}: {event_name} ({count} times)',
                        'priority': 'HIGH'
                    })
        
        return {
            'patterns': patterns,
            'recommendations': recommendations
        }
    
    def run_insights_query(self, query, hours_back=1):
        """Run CloudWatch Insights query"""
        
        end_time = datetime.now()
        start_time = end_time - timedelta(hours=hours_back)
        
        try:
            response = self.logs_client.start_query(
                logGroupName='/aws/cloudtrail',
                startTime=int(start_time.timestamp()),
                endTime=int(end_time.timestamp()),
                queryString=query
            )
            
            query_id = response['queryId']
            
            # Wait for query completion
            import time
            time.sleep(10)
            
            results = self.logs_client.get_query_results(queryId=query_id)
            return results.get('results', [])
            
        except Exception as e:
            print(f"Error running insights query: {str(e)}")
            return []

def lambda_handler(event, context):

    return {

#!/bin/bash

echo "Deploying Identity Governance Monitoring..."

# Deploy CloudFormation stack
aws cloudformation deploy \
  --template-file monitoring-infrastructure.yaml \
  --stack-name identity-governance-monitoring \
  --parameter-overrides NotificationEmail=security-team@company.com \
  --capabilities CAPABILITY_IAM

# Create Lambda functions
echo "Creating monitoring Lambda functions..."

# Package and deploy monitoring function
zip -r monitoring-function.zip monitoring_lambda.py
aws lambda create-function \
  --function-name IdentityGovernanceMonitor \
  --runtime python3.9 \
  --role arn:aws:iam::$(aws sts get-caller-identity --query Account --output text):role/IdentityGovernanceMonitoringRole \
  --handler monitoring_lambda.lambda_handler \
  --zip-file fileb://monitoring-function.zip \
  --environment Variables='{
    "LOG_GROUP_NAME":"/aws/identity-governance/events",
    "SNS_TOPIC_ARN":"arn:aws:sns:us-east-1:$(aws sts get-caller-identity --query Account --output text):IdentityGovernanceAlerts"
  }'

# Package and deploy log analyzer function
zip -r log-analyzer.zip log_analyzer.py
aws lambda create-function \
  --function-name IdentityGovernanceLogAnalyzer \
  --runtime python3.9 \
  --role arn:aws:iam::$(aws sts get-caller-identity --query Account --output text):role/IdentityGovernanceMonitoringRole \
  --handler log_analyzer.lambda_handler \
  --zip-file fileb://log-analyzer.zip

# Create EventBridge rules for scheduled monitoring
aws events put-rule \
  --name IdentityGovernanceMonitoringSchedule \
  --schedule-expression "rate(5 minutes)" \
  --description "Schedule for identity governance monitoring"

aws events put-targets \
  --rule IdentityGovernanceMonitoringSchedule \
  --targets "Id"="1","Arn"="arn:aws:lambda:us-east-1:$(aws sts get-caller-identity --query Account --output text):function:IdentityGovernanceMonitor"
