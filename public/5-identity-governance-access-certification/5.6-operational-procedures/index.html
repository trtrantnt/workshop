Bước 1: Daily Operations Procedures

1.1 Daily Monitoring Checklist

import boto3
import json
from datetime import datetime, timedelta

class DailyOperations:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.sns = boto3.client('sns')
        self.dynamodb = boto3.resource('dynamodb')
        self.operations_table = self.dynamodb.Table('OperationsLog')
    
    def execute_daily_checklist(self):
        """Execute daily operational checklist"""
        
        checklist_results = {
            'date': datetime.now().strftime('%Y-%m-%d'),
            'timestamp': datetime.now().isoformat(),
            'checks_performed': [],
            'issues_found': [],
            'actions_taken': []
        }
        
        # Daily checks
        checks = [
            ('Monitor Failed Logins', self.check_failed_logins),
            ('Review High-Risk Alerts', self.review_high_risk_alerts),
            ('Check System Health', self.check_system_health),
            ('Validate Backup Status', self.validate_backup_status),
            ('Review Pending Certifications', self.review_pending_certifications)
        ]
        
        for check_name, check_function in checks:
            try:
                result = check_function()
                checklist_results['checks_performed'].append({
                    'check': check_name,
                    'status': 'COMPLETED',
                    'result': result
                })
                
                if result.get('issues'):
                    checklist_results['issues_found'].extend(result['issues'])
                
                if result.get('actions'):
                    checklist_results['actions_taken'].extend(result['actions'])
                    
            except Exception as e:
                checklist_results['checks_performed'].append({
                    'check': check_name,
                    'status': 'FAILED',
                    'error': str(e)
                })
        
        # Log results
        self.log_operations_results(checklist_results)
        
        # Send summary if issues found
        if checklist_results['issues_found']:
            self.send_daily_summary(checklist_results)
        
        return checklist_results
    
    def check_failed_logins(self):
        """Check for unusual failed login patterns"""
        
        # Get failed login metrics from last 24 hours
        end_time = datetime.now()
        start_time = end_time - timedelta(hours=24)
        
        response = self.cloudwatch.get_metric_statistics(
            Namespace='IdentityGovernance',
            MetricName='FailedLogins',
            StartTime=start_time,
            EndTime=end_time,
            Period=3600,
            Statistics=['Sum']
        )
        
        total_failed_logins = sum(point['Sum'] for point in response['Datapoints'])
        
        result = {
            'total_failed_logins': total_failed_logins,
            'issues': [],
            'actions': []
        }
        
        # Check thresholds
        if total_failed_logins > 100:
            result['issues'].append({
                'type': 'HIGH_FAILED_LOGINS',
                'description': f'High number of failed logins: {total_failed_logins}',
                'severity': 'HIGH'
            })
            
            result['actions'].append({
                'action': 'INVESTIGATE_FAILED_LOGINS',
                'description': 'Investigate source of failed login attempts'
            })
        
        return result
    
    def review_high_risk_alerts(self):
        """Review high-risk alerts from last 24 hours"""
        
        # Query CloudWatch Logs for high-risk events
        logs_client = boto3.client('logs')
        
        end_time = datetime.now()
        start_time = end_time - timedelta(hours=24)
        
        try:
            response = logs_client.start_query(
                logGroupName='/aws/identity-governance/events',
                startTime=int(start_time.timestamp()),
                endTime=int(end_time.timestamp()),
                queryString='''
                    fields @timestamp, event_type, severity, description
                    | filter severity = "HIGH" or severity = "CRITICAL"
                    | sort @timestamp desc
                '''
            )
            
            query_id = response['queryId']
            
            # Wait for results
            import time
            time.sleep(5)
            
            results = logs_client.get_query_results(queryId=query_id)
            
            high_risk_events = len(results.get('results', []))
            
            result = {
                'high_risk_events': high_risk_events,
                'issues': [],
                'actions': []
            }
            
            if high_risk_events > 5:
                result['issues'].append({
                    'type': 'HIGH_RISK_EVENTS',
                    'description': f'Multiple high-risk events detected: {high_risk_events}',
                    'severity': 'HIGH'
                })
                
                result['actions'].append({
                    'action': 'REVIEW_HIGH_RISK_EVENTS',
                    'description': 'Review and investigate high-risk security events'
                })
            
            return result
            
        except Exception as e:
            return {
                'high_risk_events': 0,
                'issues': [{'type': 'QUERY_ERROR', 'description': str(e), 'severity': 'MEDIUM'}],
                'actions': []
            }
    
    def check_system_health(self):
        """Check overall system health"""
        
        result = {
            'system_status': 'HEALTHY',
            'issues': [],
            'actions': []
        }
        
        # Check Lambda function errors
        lambda_errors = self.check_lambda_errors()
        if lambda_errors > 10:
            result['issues'].append({
                'type': 'LAMBDA_ERRORS',
                'description': f'High Lambda error rate: {lambda_errors}',
                'severity': 'MEDIUM'
            })
            result['system_status'] = 'DEGRADED'
        
        # Check CloudWatch alarm states
        alarm_states = self.check_alarm_states()
        if alarm_states['ALARM'] > 0:
            result['issues'].append({
                'type': 'ACTIVE_ALARMS',
                'description': f'Active alarms: {alarm_states["ALARM"]}',
                'severity': 'HIGH'
            })
            result['system_status'] = 'UNHEALTHY'
        
        return result
    
    def validate_backup_status(self):
        """Validate backup and data retention status"""
        
        result = {
            'backup_status': 'OK',
            'issues': [],
            'actions': []
        }
        
        # Check S3 bucket for recent backups
        s3_client = boto3.client('s3')
        
        try:
            bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
            
            # Check for recent data
            response = s3_client.list_objects_v2(
                Bucket=bucket_name,
                Prefix=f'analytics-data/{datetime.now().strftime("%Y/%m/%d")}',
                MaxKeys=1
            )
            
            if not response.get('Contents'):
                result['issues'].append({
                    'type': 'MISSING_BACKUP',
                    'description': 'No recent backup data found',
                    'severity': 'MEDIUM'
                })
                
                result['actions'].append({
                    'action': 'INVESTIGATE_BACKUP',
                    'description': 'Investigate backup process failure'
                })
        
        except Exception as e:
            result['issues'].append({
                'type': 'BACKUP_CHECK_ERROR',
                'description': str(e),
                'severity': 'LOW'
            })
        
        return result
    
    def review_pending_certifications(self):
        """Review pending access certifications"""
        
        # Query certification table
        try:
            response = self.dynamodb.Table('CertificationTasks').scan(
                FilterExpression='#status = :status',
                ExpressionAttributeNames={'#status': 'status'},
                ExpressionAttributeValues={':status': 'PENDING'}
            )
            
            pending_count = response['Count']
            
            # Check for overdue certifications
            overdue_count = 0
            current_time = datetime.now()
            
            for item in response['Items']:
                deadline = datetime.fromisoformat(item.get('deadline', ''))
                if deadline < current_time:
                    overdue_count += 1
            
            result = {
                'pending_certifications': pending_count,
                'overdue_certifications': overdue_count,
                'issues': [],
                'actions': []
            }
            
            if overdue_count > 0:
                result['issues'].append({
                    'type': 'OVERDUE_CERTIFICATIONS',
                    'description': f'Overdue certifications: {overdue_count}',
                    'severity': 'HIGH'
                })
                
                result['actions'].append({
                    'action': 'ESCALATE_OVERDUE',
                    'description': 'Escalate overdue certifications to management'
                })
            
            return result
            
        except Exception as e:
            return {
                'pending_certifications': 0,
                'overdue_certifications': 0,
                'issues': [{'type': 'CERTIFICATION_CHECK_ERROR', 'description': str(e), 'severity': 'LOW'}],
                'actions': []
            }
    
    def log_operations_results(self, results):
        """Log operational results to DynamoDB"""
        
        try:
            self.operations_table.put_item(
                Item={
                    'operation_id': f"daily_{results['date']}",
                    'timestamp': results['timestamp'],
                    'operation_type': 'DAILY_CHECKLIST',
                    'results': json.dumps(results, default=str),
                    'issues_count': len(results['issues_found']),
                    'status': 'COMPLETED'
                }
            )
        except Exception as e:
            print(f"Error logging operations results: {str(e)}")
    
    def send_daily_summary(self, results):
        """Send daily summary if issues found"""
        
        message = f"""
        DAILY IDENTITY GOVERNANCE SUMMARY
        Date: {results['date']}
        
        CHECKS PERFORMED: {len(results['checks_performed'])}
        ISSUES FOUND: {len(results['issues_found'])}
        ACTIONS REQUIRED: {len(results['actions_taken'])}
        
        ISSUES SUMMARY:
        """
        
        for issue in results['issues_found']:
            message += f"""
        - {issue['type']}: {issue['description']} (Severity: {issue['severity']})
        """
        
        message += f"""
        
        RECOMMENDED ACTIONS:
        """
        
        for action in results['actions_taken']:
            message += f"""
        - {action['action']}: {action['description']}
        """
        
        try:
            self.sns.publish(
                TopicArn='arn:aws:sns:region:account:identity-governance-operations',
                Subject=f'Daily Identity Governance Summary - {results["date"]}',
                Message=message
            )
        except Exception as e:
            print(f"Error sending daily summary: {str(e)}")

def lambda_handler(event, context):
    """Lambda handler for daily operations"""
    
    operations = DailyOperations()
    results = operations.execute_daily_checklist()
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Daily operations completed',
            'date': results['date'],
            'checks_performed': len(results['checks_performed']),
            'issues_found': len(results['issues_found']),
            'actions_taken': len(results['actions_taken'])
        })
    }

1.2 Standard Operating Procedures (SOPs)

# Identity Governance Standard Operating Procedures

## SOP-001: Daily Monitoring Procedures

### Purpose
Ensure daily monitoring of identity governance systems and prompt response to issues.

### Scope
All identity governance monitoring activities performed daily.

### Procedure
1. **Morning Health Check (9:00 AM)**
   - Review CloudWatch dashboard
   - Check for active alarms
   - Verify system availability

2. **Failed Login Review (10:00 AM)**
   - Analyze failed login patterns
   - Investigate suspicious activities
   - Document findings

3. **Risk Alert Review (2:00 PM)**
   - Review high-risk alerts
   - Prioritize investigations
   - Assign remediation tasks

4. **End-of-Day Summary (5:00 PM)**
   - Generate daily report
   - Update incident tracking
   - Plan next day activities

### Escalation Criteria
- Critical alarms: Immediate escalation
- High-risk events: 2-hour response time
- System outages: Immediate escalation

## SOP-002: Incident Response Procedures

### Purpose
Standardize response to identity governance security incidents.

### Incident Classification
- **P1 (Critical)**: System compromise, data breach
- **P2 (High)**: Privilege escalation, unauthorized access
- **P3 (Medium)**: Policy violations, anomalous behavior
- **P4 (Low)**: Minor configuration issues

### Response Procedures

#### P1 - Critical Incidents
1. **Immediate Actions (0-15 minutes)**
   - Activate incident response team
   - Isolate affected systems
   - Preserve evidence
   - Notify stakeholders

2. **Investigation (15-60 minutes)**
   - Analyze logs and evidence
   - Determine scope of impact
   - Identify root cause
   - Document findings

3. **Containment (1-4 hours)**
   - Implement containment measures
   - Revoke compromised credentials
   - Apply security patches
   - Monitor for persistence

4. **Recovery (4-24 hours)**
   - Restore normal operations
   - Validate system integrity
   - Update security controls
   - Conduct lessons learned

#### P2 - High Incidents
1. **Response Time**: 2 hours
2. **Investigation**: 4 hours
3. **Resolution**: 24 hours

### Documentation Requirements
- Incident timeline
- Actions taken
- Evidence collected
- Lessons learned
- Preventive measures

## SOP-003: Access Certification Procedures

### Purpose
Standardize quarterly access certification process.

### Pre-Certification (Week 1)
1. **Data Collection**
   - Extract current access assignments
   - Identify certification owners
   - Prepare certification packages

2. **Notification**
   - Send certification requests
   - Provide instructions and deadlines
   - Set up tracking system

### Certification Period (Weeks 2-4)
1. **Monitoring**
   - Track response rates
   - Send reminders
   - Escalate overdue items

2. **Support**
   - Answer questions
   - Provide technical assistance
   - Resolve issues

### Post-Certification (Week 5)
1. **Analysis**
   - Compile results
   - Identify denied access
   - Generate remediation tasks

2. **Remediation**
   - Execute access removals
   - Update documentation
   - Validate changes

## SOP-004: Change Management Procedures

### Purpose
Control changes to identity governance systems.

### Change Categories
- **Emergency**: Security-related, immediate implementation
- **Standard**: Pre-approved, low-risk changes
- **Normal**: Requires approval, scheduled implementation

### Approval Process
1. **Change Request Submission**
   - Complete change request form
   - Include risk assessment
   - Specify rollback plan

2. **Review and Approval**
   - Technical review
   - Security assessment
   - Management approval

3. **Implementation**
   - Schedule maintenance window
   - Execute change procedures
   - Validate results

4. **Post-Implementation**
   - Monitor for issues
   - Update documentation
   - Close change request

### Emergency Change Process
1. **Immediate Implementation** (if required)
2. **Notification** within 1 hour
3. **Retrospective Review** within 24 hours
4. **Documentation Update** within 48 hours

2.1 Weekly Assessment Script

import boto3
import json
from datetime import datetime, timedelta

class WeeklyReview:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.s3_client = boto3.client('s3')
        self.athena_client = boto3.client('athena')
    
    def execute_weekly_review(self):
        """Execute weekly identity governance review"""
        
        review_results = {
            'week_ending': datetime.now().strftime('%Y-%m-%d'),
            'timestamp': datetime.now().isoformat(),
            'metrics_summary': {},
            'trend_analysis': {},
            'recommendations': []
        }
        
        # Collect weekly metrics
        review_results['metrics_summary'] = self.collect_weekly_metrics()
        
        # Perform trend analysis
        review_results['trend_analysis'] = self.analyze_trends()
        
        # Generate recommendations
        review_results['recommendations'] = self.generate_weekly_recommendations()
        
        # Store results
        self.store_weekly_results(review_results)
        
        return review_results
    
    def collect_weekly_metrics(self):
        """Collect metrics for the past week"""
        
        end_time = datetime.now()
        start_time = end_time - timedelta(days=7)
        
        metrics = {}
        
        # Failed logins
        metrics['failed_logins'] = self.get_metric_sum(
            'FailedLogins', start_time, end_time
        )
        
        # Successful logins
        metrics['successful_logins'] = self.get_metric_sum(
            'SuccessfulLogins', start_time, end_time
        )
        
        # Permission changes
        metrics['permission_changes'] = self.get_metric_sum(
            'PermissionChanges', start_time, end_time
        )
        
        # High-risk events
        metrics['high_risk_events'] = self.get_metric_sum(
            'HighRiskEvents', start_time, end_time
        )
        
        # Certification completions
        metrics['certifications_completed'] = self.get_certification_metrics()
        
        return metrics
    
    def analyze_trends(self):
        """Analyze trends over the past 4 weeks"""
        
        trends = {}
        
        # Get data for past 4 weeks
        for week in range(4):
            end_time = datetime.now() - timedelta(days=week*7)
            start_time = end_time - timedelta(days=7)
            
            week_data = {
                'failed_logins': self.get_metric_sum('FailedLogins', start_time, end_time),
                'permission_changes': self.get_metric_sum('PermissionChanges', start_time, end_time),
                'high_risk_events': self.get_metric_sum('HighRiskEvents', start_time, end_time)
            }
            
            trends[f'week_{4-week}'] = week_data
        
        # Calculate trend direction
        trend_analysis = {}
        for metric in ['failed_logins', 'permission_changes', 'high_risk_events']:
            values = [trends[f'week_{i}'][metric] for i in range(1, 5)]
            trend_analysis[metric] = self.calculate_trend(values)
        
        return trend_analysis
    
    def calculate_trend(self, values):
        """Calculate trend direction (increasing, decreasing, stable)"""
        
        if len(values) < 2:
            return 'insufficient_data'
        
        # Simple trend calculation
        recent_avg = sum(values[-2:]) / 2
        older_avg = sum(values[:-2]) / len(values[:-2]) if len(values) > 2 else values[0]
        
        change_percent = ((recent_avg - older_avg) / older_avg * 100) if older_avg > 0 else 0
        
        if change_percent > 20:
            return 'increasing'
        elif change_percent < -20:
            return 'decreasing'
        else:
            return 'stable'
    
    def generate_weekly_recommendations(self):
        """Generate recommendations based on weekly analysis"""
        
        recommendations = []
        
        # Get current metrics
        metrics = self.collect_weekly_metrics()
        trends = self.analyze_trends()
        
        # Failed login recommendations
        if metrics['failed_logins'] > 500:
            recommendations.append({
                'category': 'SECURITY',
                'priority': 'HIGH',
                'title': 'High Failed Login Volume',
                'description': f"Failed logins this week: {metrics['failed_logins']}",
                'action': 'Investigate failed login patterns and consider implementing additional security measures'
            })
        
        # Trend-based recommendations
        if trends['high_risk_events'] == 'increasing':
            recommendations.append({
                'category': 'RISK',
                'priority': 'HIGH',
                'title': 'Increasing High-Risk Events',
                'description': 'High-risk events are trending upward',
                'action': 'Review security controls and consider additional monitoring'
            })
        
        # Permission change recommendations
        if metrics['permission_changes'] > 100:
            recommendations.append({
                'category': 'GOVERNANCE',
                'priority': 'MEDIUM',
                'title': 'High Permission Change Volume',
                'description': f"Permission changes this week: {metrics['permission_changes']}",
                'action': 'Review permission change patterns for compliance'
            })
        
        return recommendations
    
    def get_metric_sum(self, metric_name, start_time, end_time):
        """Get sum of metric values over time period"""
        
        try:
            response = self.cloudwatch.get_metric_statistics(
                Namespace='IdentityGovernance',
                MetricName=metric_name,
                StartTime=start_time,
                EndTime=end_time,
                Period=3600,
                Statistics=['Sum']
            )
            
            return sum(point['Sum'] for point in response['Datapoints'])
            
        except Exception as e:
            print(f"Error getting metric {metric_name}: {str(e)}")
            return 0
    
    def store_weekly_results(self, results):
        """Store weekly review results"""
        
        bucket_name = f'privilege-analytics-{boto3.Session().region_name}'
        key = f'weekly-reviews/{datetime.now().strftime("%Y/%m")}/week-ending-{results["week_ending"]}.json'
        
        try:
            self.s3_client.put_object(
                Bucket=bucket_name,
                Key=key,
                Body=json.dumps(results, indent=2, default=str),
                ContentType='application/json'
            )
            
            print(f"Weekly review stored: s3://{bucket_name}/{key}")
            
        except Exception as e:
            print(f"Error storing weekly results: {str(e)}")

def lambda_handler(event, context):
    """Lambda handler for weekly review"""
    
    review = WeeklyReview()
    results = review.execute_weekly_review()
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Weekly review completed',
            'week_ending': results['week_ending'],
            'recommendations': len(results['recommendations'])
        })
    }
